\begin{equation*}
  \begin{aligned}
    x_o = A x_i \quad\quad x_f = \beta x_o \quad\quad x_i = x_s - x_f
  \end{aligned}
\end{equation*}

\begin{equation*}
  \begin{aligned}
    A_f = \dfrac{x_0}{x_s}  = \dfrac{A x_i}{x_i + \beta A x_i } = \dfrac{A}{1 + A \beta} \approx \dfrac{1}{\beta}  \quad\quad \left( A\beta \gg 1 \right)
  \end{aligned}
\end{equation*}
