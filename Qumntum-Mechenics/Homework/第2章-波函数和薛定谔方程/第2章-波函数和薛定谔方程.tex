\paragraph{2.2}

\begin{equation*}
  \begin{aligned}
    \psi_1 = \dfrac{1}{r} \exp \left[ i k r \right]
    \quad\quad \quad\quad \quad\quad 
    \psi_2 = \dfrac{1}{r} \exp \left[ - i k r \right] 
  \end{aligned}
\end{equation*}
